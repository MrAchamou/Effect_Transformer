## 🤖 **Algorithme de Détection**

```javascript
function detectAvailableLevels(effectName, category, subcategory) {
  
  // Niveau 1 seulement - Effets UI/UX et transitions simples
  if (category === 'ui_ux' || isSimpleTransition(subcategory)) {
    return { levels: [1], reason: "Effet UI optimisé" };
  }
  
  // Niveaux 1+2 - Texte, images, audio
  if (['texte', 'images', 'audio'].includes(category) && 
      !isAdvanced3D(subcategory)) {
    return { levels: [1, 2], reason: "Potentiel modéré d'amélioration" };
  }
  
  // Tous niveaux - Particules, 3D, vidéo complexe
  if (['particules_simulation', '3d_motion', 'video_avance'].includes(category)) {
    return { levels: [1, 2, 3], reason: "Fort potentiel d'amélioration IA" };
  }
  
  // Fallback sécurisé
  return { levels: [1, 2], reason: "Classification par défaut" };
}
```
